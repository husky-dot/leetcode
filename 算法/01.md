盛最多水的容器

```
var maxArea = function(height) {
    var i = 0, j = height.length - 1, max = 0
    while (i < j) {
        max = Math.max(max, (j - i) * (height[i] > height[j] ? height[j--]: height[i++]))
    }
    return max
};
```

 移动零

 ```
 var moveZeroes = function(nums) {
    let j = 0
    for (let i = 0; i < nums.length; i++) {
        if (nums[i] !==0 ) {
            nums[j] = nums[i]
            if (i !== j) {
                nums[i] = 0
            }
            j++
        }
    }
};
 ```
