69. x 的平方根

https://leetcode-cn.com/problems/sqrtx/
```
var mySqrt = function(x) {
    let left = 0, right = x ,ans = -1;

    while (left <= right) {
        let mid = Math.floor (left + (right - left) / 2)
        if (mid * mid <= x) {
            ans = mid
            left = mid + 1
        } else {
            right = mid - 1
        }
    }

    return ans;
};
```
