function mergeTwoLists( l1 ,  l2 ) {
    // write code here
    if (!l1) {
        return l2
    }
    if (!l2) {
        return l1
    }
    let dummy = new ListNode(-1);
    let curr = dummy;
    
    while (l1 && l2) {
        if (l1.val <= l2.val) {
            curr.next = l1
            l1 = l1.next;
            
        } else {
            curr.next = l2
            l2 = l2.next
        }
        curr = curr.next
    }
    if (l1) {
        curr.next = l1
    }
    if (l2) {
        curr.next = l2
    }
    return dummy.next
}

```
function maxLength( arr ) {
    const set = new Set()
    const len = arr.length;
    let rk = -1, ans = 0;
    
    for (let i = 0; i < len; i++) {
        if (i!=0) {
            set.delete(arr[i])
        }
        while(rk + 1 < len && !set.has(arr[rk + 1])) {
            set.add(arr[rk + 1])
            rk++
        }
        ans = Math.max(ans, rk - i)
    }
    return ans;
}
```

var removeNthFromEnd = function(head, n) {
    let preHead = new ListNode(0)
    preHead.next = head
    let fast = preHead, slow = preHead
    // 快先走 n+1 步
    while(n--) {
        fast = fast.next
    }
    // fast、slow 一起前进
    while(fast && fast.next) {
        fast = fast.next
        slow = slow.next
    }
    slow.next = slow.next.next
    return preHead.next
};
